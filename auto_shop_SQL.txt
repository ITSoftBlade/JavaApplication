CREATE TABLE CUSTUMERS( 
ID_CUSTUMER INT NOT NULL AUTO_INCREMENT, 
FIRST_NAME VARCHAR(25) NOT NULL, 
LAST_NAME VARCHAR(25) NOT NULL, 
EMAIL VARCHAR(30) NOT NULL UNIQUE, 
PHONE INT NULL, 
ADDRESS VARCHAR(40) NULL,
PRIMARY KEY (ID_CUSTUMER)
);

CREATE TABLE CARS (
ID_CAR INT NOT NULL AUTO_INCREMENT, 
MAKE VARCHAR(20) NOT NULL,
MODEL VARCHAR(20) NOT NULL, 
NO VARCHAR(20) NULL UNIQUE, 
PRICE DECIMAL(10, 2) NOT NULL, 
YEAR INT NOT NULL,
PRIMARY KEY (ID_CAR)
);

CREATE TABLE ORDERS ( ID_ORDER INT NOT NULL, ID_CUSTUMER INT NOT NULL, ID_CAR INT NOT NULL, PRIMARY KEY (ID_ORDER), FOREIGN KEY(ID_CUSTUMER) REFERENCES CUSTUMERS (ID_CUSTUMER), FOREIGN KEY(ID_CAR) REFERENCES CARS (ID_CAR));
