CREATE TABLE CUSTUMERS( 
ID_CUSTUMER INT NOT NULL AUTO_INCREMENT, 
FIRST_NAME VARCHAR(25) NOT NULL, 
LAST_NAME VARCHAR(25) NOT NULL, 
EMAIL VARCHAR(30) NOT NULL, 
PHONE INT NULL, 
ADDRESS VARCHAR(40) NULL,
PRIMARY KEY (ID_CUSTUMER);

CREATE TABLE MAKES (
ID_MAKE INT NOT NULL AUTO_INCREMENT,
MAKE VARCHAR(20) NOT NULL,
MODEL VARCHAR(20) NOT NULL UNIQUE,
PRIMARY KEY (ID_MAKE);

CREATE TABLE CARS (
ID_CAR INT NOT NULL AUTO_INCREMENT, 
ID_MAKE INT NOT NULL, 
NO VARCHAR(20) NULL, 
PRICE DECIMAL(10, 2) NOT NULL, 
YEAR INT NOT NULL,
PRIMARY KEY (ID_CAR , ID_MAKE),
FOREIGN KEY(ID_MAKE) REFERENCES MAKES(ID_MAKE),
);

CREATE TABLE ORDERS (
ID_ORDER INT NOT NULL AUTO_INCREMENT, 
ID_CUSTUMER INT NOT NULL, 
ID_CAR INT NOT NULL, 
DATA TIMESTAMP NOT NULL,
PRIMARY KEY ( ID_ORDER , ID_CUSTUMER , ID_CAR ),
FOREIGN KEY(ID_CUSTUMER) REFERENCES CUSTUMERS(ID_CUSTUMER),
FOREIGN KEY(ID_CAR) REFERENCES CARS(ID_CAR)
);
