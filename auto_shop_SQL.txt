CREATE TABLE CUSTUMERS( 
ID_CUSTUMER INT PRIMARY KEY, 
FIRST_NAME VARCHAR(25) NOT NULL, 
LAST_NAME VARCHAR(25) NOT NULL, 
EMAIL VARCHAR(30) NOT NULL, 
PHONE INT NULL, 
ADDRESS VARCHAR(40) NULL 
);

CREATE TABLE MODELS (
ID_MODEL INT PRIMARY KEY, 
MODEL VARCHAR(20) NOT NULL);

CREATE TABLE MAKES (
ID_MAKE INT PRIMARY KEY,
MAKE VARCHAR(20) NOT NULL);

CREATE TABLE CARS (
ID_CAR INT NOT NULL, 
ID_MAKE INT NOT NULL,
ID_MODEL INT NOT NULL, 
NO VARCHAR(20) NULL, 
PRICE DECIMAL(10, 2) NOT NULL, 
YEAR INT NOT NULL 
);

CREATE TABLE ORDERS (
ID_ORDER INT NOT NULL, 
ID_CUSTUMER INT NOT NULL, 
ID_CAR INT NOT NULL, 
DATA TIMESTAMP NOT NULL 
);

ALTER TABLE ORDERS ADD CONSTRAINT PRIMARY KEY ( ID_ORDER , ID_CUSTUMER , ID_CAR );
ALTER TABLE CARS ADD CONSTRAINT PRIMARY KEY ( ID_CAR , ID_MAKE , ID_MODEL );
